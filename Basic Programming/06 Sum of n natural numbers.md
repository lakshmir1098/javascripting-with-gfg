1. Using for Loop
    ```js
    function findSum(n) { 
        let sum = 0; 
        for (let i = 1; i <= n; i++) 
            sum = sum + i; 
        return sum; 
    } 
    ```
2. Using Recursion
    ```js
    function findSum(n) { 
        let sum = 0; 
        for (let i = 1; i <= n; i++) 
            sum = sum + i; 
        return sum; 
    } 
    ```
3. Using Mathematical formula
    ```js
    function findSum(n) { 
        return n * (n + 1) / 2; 
    } 
    ```
